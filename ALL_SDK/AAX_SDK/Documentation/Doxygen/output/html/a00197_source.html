177  void Add(const AAX_CString& inAbbreviation)
178  {
179  uint32_t stringLength = inAbbreviation.Length();
180  mAbbreviations[stringLength] = inAbbreviation; //Does a string copy into the map.
181  }
182 
183  const AAX_CString& Get(int32_t inNumCharacters) const
184  {
185  //More characters than the primary string or no specific shortened names.
186  if ((inNumCharacters >= int32_t(mPrimary.Length())) || (mAbbreviations.empty()) || (0 > inNumCharacters))
187  return mPrimary;
188 
189  std::map<uint32_t, AAX_CString>::const_iterator iter = mAbbreviations.upper_bound(static_cast<uint32_t>(inNumCharacters));
190 
191  //If the iterator is already pointing to shortest string, return that.
192  if (iter == mAbbreviations.begin())
193  return iter->second;
194 
195  //lower_bound() will return the iterator that is larger than the desired value, so decrement the iterator.
196  --iter;
197  return iter->second;
198  }
199 
200  void Clear() { mAbbreviations.clear(); }
201 
202 private:
203  AAX_CString mPrimary;
204  std::map<uint32_t, AAX_CString> mAbbreviations;
205 };

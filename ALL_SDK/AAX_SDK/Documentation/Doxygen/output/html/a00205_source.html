70 inline
71 void
72 AAX_EndianSwapInPlace(T* theDataP)
73 {
74  char *begin, *end;
75 
76  begin = reinterpret_cast<char*>(theDataP);
77  end = begin + sizeof( T );
78  std::reverse( begin, end );
79 }

416 inline AAX_CBoolean sampleRateInMask(AAX_CSampleRate inSR, uint32_t iMask)
417 {
418  return static_cast<AAX_CBoolean>(
419  (44100.0 == inSR) ? ((iMask & AAX_eSampleRateMask_44100) != 0) :
420  (48000.0 == inSR) ? ((iMask & AAX_eSampleRateMask_48000) != 0) :
421  (88200.0 == inSR) ? ((iMask & AAX_eSampleRateMask_88200) != 0) :
422  (96000.0 == inSR) ? ((iMask & AAX_eSampleRateMask_96000) != 0) :
423  (176400.0 == inSR) ? ((iMask & AAX_eSampleRateMask_176400) != 0) :
424  (192000.0 == inSR) ? ((iMask & AAX_eSampleRateMask_192000) != 0) : false
425  );
426 }

437  ((iMask & AAX_eSampleRateMask_88200) != 0) ? 88200.0f :

439  ((iMask & AAX_eSampleRateMask_176400) != 0) ? 176400.0f :

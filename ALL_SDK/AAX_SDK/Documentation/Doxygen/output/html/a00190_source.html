625 template<typename T>
626 void AAX_CParameter<T>::SetValue( T newValue )
627 {
628  double newNormalizedValue = mTaperDelegate->RealToNormalized(newValue);
629 
630  // <DMT> Always go through the automation delegate even if the control isn't automatable to prevent fighting with other GUIs.
631  // Somewhere back in the automation delegate, or elsewhere in the system, it will determine the differences in behavior surrounding
632  // automation. The only reason that there wouldn't be an automation delegate is if this parameter has yet to be added to a
633  // ParameterManager. Let's put the null value guards in place, just in case, and also for unit tests.
634  if ( mAutomationDelegate )
635  {
636  //TODO: Create RAII utility class for touch/release
637 
638  //Touch the control
639  Touch();
640 
641  //Send that token.
642  mAutomationDelegate->PostSetValueRequest(Identifier(), newNormalizedValue );
643 
644  //Release the control
645  Release();
646  }
647  else
648  {
649  mNeedNotify = true;
650 
651  // In the rare case that an automation delegate doesn't exist, lets still set the value. It's possible that someone is trying to
652  // set the new value before adding the parameter to a parametermanager.
653  UpdateNormalizedValue(newNormalizedValue);
654  }
655 }

657 template <typename T>
658 void AAX_CParameter<T>::UpdateNormalizedValue(double newNormalizedValue)
659 {
660  T newValue = mTaperDelegate->NormalizedToReal(newNormalizedValue);
661  if (mNeedNotify || (mValue.Get() != newValue))
662  {
663  //Set the new value
664  mValue.Set(newValue);
665 
666  //<DMT> Always notify that the value has changed through the automation delegate to guarantee that all control surfaces and other
667  // GUIs get their values updated.
668  if (mAutomationDelegate)
669  mAutomationDelegate->PostCurrentValue(Identifier(), newNormalizedValue);
670 
671  // clear flag
672  mNeedNotify = false;
673  }

676 template <typename T>
677 void AAX_CParameter<T>::InitializeNumberOfSteps()
678 {
679  if (mNumSteps == 0) // If no explicit number of steps has been set...
680  {
681  switch (mControlType)
682  {
683  case AAX_eParameterType_Discrete:
684  {
685  // Discrete parameters default to binary unless
686  // otherwise specified
687  this->SetNumberOfSteps (eParameterDefaultNumStepsDiscrete);
688  break;
689  }
690  case AAX_eParameterType_Continuous:
691  {
692  // Defaulting to 128 steps to match one full rotation of
693  // Command|8 and similar surfaces, which query the num
694  // steps to determine tick values for rotary encoders
695  this->SetNumberOfSteps (eParameterDefaultNumStepsContinuous);
696  break;
697  }
698  default:
699  {
700  AAX_ASSERT (0); // Invalid type
701  break;
702  }
703  }
704  }
